-- Создание таблицы "Страна"
CREATE TABLE Country (
    id_country SERIAL PRIMARY KEY, -- Id страны (PK)
    name_country VARCHAR(70) NOT NULL, -- Название страны (максимальная длина 70)
    code_country VARCHAR(10), -- Код страны
    capital VARCHAR(70), -- Столица (максимальная длина 70)
    continent VARCHAR(20) CHECK (continent IN ('Евразия', 'Северная Америка', 'Южная Америка', 'Австралия', 'Антарктида', 'Африка')), -- Континент
    official_language TEXT, -- Официальный язык
    currency TEXT, -- Валюта
    time_zone VARCHAR(50) -- Часовой пояс
);

-- Создание таблицы "Инвестиционный проект"
CREATE TABLE Investment_Project (
    id_ip SERIAL PRIMARY KEY, -- Id ИП (PK)
    id_country INTEGER NOT NULL, -- Id страны (FK)
    full_ip_name TEXT NOT NULL, -- Полное название ИП
    short_ip_name TEXT, -- Краткое название ИП
    goal_ip TEXT, -- Цель ИП
    code_ip VARCHAR(40), -- Код ИП (текст с числами)
    start_date_ip INTEGER, -- Стартовая дата
    end_date_ip INTEGER, -- Ожидаемая дата завершения
    decision_level TEXT, -- Уровень принятия решения
    significance_flag BOOLEAN, -- Флаг значимости (true/false)
    FOREIGN KEY (id_country) REFERENCES Country(id_country) -- Внешний ключ к таблице Страна

-- Создание таблицы "Комплекс"
CREATE TABLE Complex (
    id_complex SERIAL PRIMARY KEY, -- Id комплекса (PK)
    full_complex_name TEXT NOT NULL,        -- Полное название комплекса
    short_complex_name TEXT,       -- Краткое название комплекса
    area_complex_km2 NUMERIC,              -- Площадь комплекса, км?
    address_complex TEXT                    -- Адрес комплекса
);

-- Создание таблицы "Объект"
CREATE TABLE Object (
    id_object SERIAL PRIMARY KEY, -- Id объекта (PK)
    id_ip INTEGER, -- Id ИП (FK)
    id_region INTEGER NOT NULL, -- Id региона (FK)
    id_division INTEGER NOT NULL, -- Id дивизиона (FK)
    id_complex INTEGER, -- Id комплекса (FK)
    id_main_object INTEGER, -- Id главного объекта (FK)
    code_by_classifier CHAR(14) NOT NULL, -- Код по классификатору (текст в формате трехзначного числа)
    full_object_name TEXT NOT NULL, -- Полное название объекта
    short_object_name TEXT, -- Краткое название объекта
    visibility_level INTEGER CHECK (visibility_level BETWEEN 1 AND 4), -- Уровень видимости (число от 1 до 4)
    security_classification INTEGER CHECK (security_classification BETWEEN 1 AND 4), -- Гриф секретности (число от 1 до 4)
    address_object TEXT, -- Адрес объекта
    area_object_km2 NUMERIC, -- Площадь объекта, км2
    FOREIGN KEY (id_division) REFERENCES Division(id_division), -- Внешний ключ к таблице Division
    FOREIGN KEY (id_region) REFERENCES Region(id_region), -- Внешний ключ к таблице Region
    FOREIGN KEY (id_complex) REFERENCES Complex(id_complex), -- Внешний ключ к таблице Complex
    FOREIGN KEY (id_object) REFERENCES Object(id_object) -- Внешний ключ к таблице Object
);

    month VARCHAR(20) CHECK (month IN ('Январь', 'Февраль', 'Март', 'Апрель', 'Май', 'Июнь', 'Июль', 'Август', 'Сентябрь', 'Октябрь', 'Ноябрь', 'Декабрь')), -- Месяц
    year INTEGER, -- Год (число целое)
    fact_or_plan VARCHAR(10) CHECK (fact_or_plan IN ('факт', 'план')), -- Факт или план
    cumulative_effect BOOLEAN NOT NULL DEFAULT TRUE, -- Накопительный эффект (флаг)
    FOREIGN KEY (id_ip) REFERENCES Investment_Project(id_ip), -- Внешний ключ к таблице "Инвестиционный проект"
    FOREIGN KEY (id_object) REFERENCES Object(id_object), -- Внешний ключ к таблице "Объект"
    FOREIGN KEY (id_category) REFERENCES Category_Indicators(id_category), -- Внешний ключ к таблице "Категория показателей"
    FOREIGN KEY (id_expense_direction) REFERENCES Expense_Direction(id_expense_direction) -- Внешний ключ к таблице "Направление затрат"
);
